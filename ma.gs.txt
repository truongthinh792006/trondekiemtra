function doGet() {
  return HtmlService.createTemplateFromFile('Index')
      .evaluate()
      .setTitle('Hệ Thống Trộn Đề Thi Thông Minh')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
      .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

/**
 * LÕI ĐIỀU KHIỂN ĐỊNH DẠNG TỰ ĐỘNG CĂNG ĐỀU (FONT: TIMES NEW ROMAN):
 * - ĐỀ GỐC/ĐÁP ÁN: 100% xuống hàng dọc thoáng đãng, liên tục.
 * - ĐỀ TRỘN (ZIP): Tự động co dòng linh hoạt 1 dòng (bảng ẩn 1x4 căng đều) -> 2 dòng (bảng ẩn 2x2) -> 4 dòng dọc.
 */
function handleExportRequest(questions, codeCount, rawOriginalText, mode) {
  try {
    if (!questions || questions.length === 0) return { success: false, error: "Chưa có dữ liệu câu hỏi!" };
    
    var startCode = 101;
    var temporaryFileIds = [];
    
    function cleanTextData(str) {
      if (!str) return "";
      return str.replace(/\s*đáp\s*án\s*[:\-]*\s*.*$/gi, "").trim();
    }

    function convertToDS(ansChar) {
      if (ansChar === 'Đ' || ansChar === 'D' || ansChar === 'Đúng' || ansChar === 'True' || ansChar === 'T') return 'D';
      return 'S';
    }

    function appendOrReuseParagraph(body, text) {
      var numChildren = body.getNumChildren();
      if (numChildren > 0) {
        var lastChild = body.getChild(numChildren - 1);
        if (lastChild.getType() === DocumentApp.ElementType.PARAGRAPH) {
          var lastPara = lastChild.asParagraph();
          if (lastPara.getText() === "") {
            lastPara.setText(text);
            return lastPara;
          }
        }
      }
      return body.appendParagraph(text);
    }

    // HÀM RENDER CÂU HỎI THÔNG MINH TỰ ĐỘNG CO DÒNG LINH HOẠT VÀ CĂNG ĐỀU
    function renderQuestionBlock(q, displayId, includeAnswers, isCompactMode, body) {
      var bodyLines = q.body ? q.body.split('\n') : [];
      while (bodyLines.length > 0 && bodyLines[0].trim() === '') {
        bodyLines.shift();
      }
      while (bodyLines.length > 0 && bodyLines[bodyLines.length - 1].trim() === '') {
        bodyLines.pop();
      }
      
      var validLines = [];
      bodyLines.forEach(function(line) {
        if (isCompactMode && line.trim() === '') return; 
        validLines.push(line);
      });

      // Xuất thân câu hỏi
      validLines.forEach(function(line, lIdx) {
        var prefix = lIdx === 0 ? ('Câu ' + displayId + ': ') : '';
        var qPara = appendOrReuseParagraph(body, prefix + line);
        qPara.setFontFamily('Times New Roman').setFontSize(11).setBold(true);
        qPara.setLineSpacing(1.0).setSpacingAfter(0);
        
        if (lIdx === 0) {
          qPara.setSpacingBefore(!isCompactMode && displayId > 1 ? 12 : 0);
        } else {
          qPara.setSpacingBefore(0);
        }
        
        if (lIdx > 0 && (line.indexOf('    ') === 0 || line.indexOf('\t') === 0 || line.indexOf('for ') !== -1 || line.indexOf('print') !== -1)) {
          qPara.setIndentStart(24);
        }
      });
      
      // XỬ LÝ PHƯƠNG ÁN LỰA CHỌN (MCQ)
      if (q.type === 'MCQ') {
        var cleanOptions = q.options.map(function(opt) {
          return cleanTextData(opt).trim();
        });

        // NẾU LÀ FILE TRỘN TRONG ZIP -> KÍCH HOẠT THUẬT TOÁN TỰ ĐỘNG CO DÒNG & CĂNG ĐỀU TRÊN TRANG GIẤY
        if (isCompactMode) {
          var lenA = cleanOptions[0].length;
          var lenB = cleanOptions[1].length;
          var lenC = cleanOptions[2].length;
          var lenD = cleanOptions[3].length;
          var maxLen = Math.max(lenA, lenB, lenC, lenD);
          var totalLen = lenA + lenB + lenC + lenD;

          // KỊCH BẢN 1: TẤT CẢ PHƯƠNG ÁN ĐỀU NGẮN -> CHÈN VÀO BẢNG ẨN 1 HÀNG 4 CỘT ĐỂ CĂNG ĐỀU TOÀN TRANG
          if (totalLen <= 55 && maxLen <= 15) {
            var tableCells = [
              ['   A. ' + cleanOptions[0], '   B. ' + cleanOptions[1], '   C. ' + cleanOptions[2], '   D. ' + cleanOptions[3]]
            ];
            var optTable = body.appendTable(tableCells);
            optTable.setBorderColor('#FFFFFF'); // Ẩn viền hoàn toàn để trông như chữ viết bình thường
            
            var row = optTable.getRow(0);
            for (var c = 0; c < 4; c++) {
              var cell = row.getCell(c);
              cell.setPaddingTop(0).setPaddingBottom(1).setPaddingLeft(0).setPaddingRight(0);
              var p = cell.getChild(0).asParagraph();
              p.setFontSize(11).setFontFamily('Times New Roman').setBold(false);
              p.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
            }
          }
          // KỊCH BẢN 2: ĐỘ DÀI VỪA PHẢI -> CHIA THÀNH 2 DÒNG (THẲNG HÀNG CỘT B & D BẰNG KHUNG ẨN 2x2)
          else if (totalLen <= 120 && maxLen <= 30) {
            var tableCells = [
              ['   A. ' + cleanOptions[0], '   B. ' + cleanOptions[1]],
              ['   C. ' + cleanOptions[2], '   D. ' + cleanOptions[3]]
            ];
            var optTable = body.appendTable(tableCells);
            optTable.setBorderColor('#FFFFFF'); // Ẩn viền hoàn toàn
            
            for (var r = 0; r < 2; r++) {
              var row = optTable.getRow(r);
              for (var c = 0; c < 2; c++) {
                var cell = row.getCell(c);
                cell.setPaddingTop(0).setPaddingBottom(1).setPaddingLeft(0).setPaddingRight(0);
                var p = cell.getChild(0).asParagraph();
                p.setFontSize(11).setFontFamily('Times New Roman').setBold(false);
                p.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
              }
            }
          }
          // KỊCH BẢN 3: ĐÁP ÁN DÀI HOẶC CHỨA CODE -> XUỐNG 4 DÒNG DỌC BIÊN ĐỨNG
          else {
            cleanOptions.forEach(function(opt, oIdx) {
              var label = String.fromCharCode(65 + oIdx);
              var optPara = appendOrReuseParagraph(body, '   ' + label + '. ' + opt);
              optPara.setFontSize(11).setFontFamily('Times New Roman').setBold(false).setIndentStart(36);
              optPara.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
            });
          }
          
          // Ghi đáp án cho đề trộn (nếu chế độ bật) bám sát ngay bên dưới
          if (includeAnswers) {
            var ansLetter = q.correctAnswer !== null ? String.fromCharCode(65 + q.correctAnswer) : "A";
            var ansPara = appendOrReuseParagraph(body, '   Đáp án: ' + ansLetter);
            ansPara.setFontSize(11).setFontFamily('Times New Roman').setBold(true).setIndentStart(36).setForegroundColor('#000000');
            ansPara.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
          }
        } 
        // CHẾ ĐỘ ĐỀ GỐC & ĐỀ ĐÁP ÁN -> LUÔN GIỮ NGUYÊN TẮC XUỐNG 4 DÒNG DỌC THOÁNG ĐÃNG
        else {
          cleanOptions.forEach(function(opt, oIdx) {
            var label = String.fromCharCode(65 + oIdx);
            var optPara = appendOrReuseParagraph(body, '   ' + label + '. ' + opt);
            optPara.setFontSize(11).setFontFamily('Times New Roman').setBold(false).setIndentStart(36);
            optPara.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
          });
          
          if (includeAnswers) {
            var ansLetter = q.correctAnswer !== null ? String.fromCharCode(65 + q.correctAnswer) : "A";
            var ansPara = appendOrReuseParagraph(body, '   Đáp án: ' + ansLetter);
            ansPara.setFontSize(11).setFontFamily('Times New Roman').setBold(true).setIndentStart(36).setForegroundColor('#000000');
            ansPara.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
          }
        }
      } 
      // Xử lý câu hỏi Đúng/Sai (TF)
      else if (q.type === 'TF') {
        q.subQuestions.forEach(function(sub) {
          var optPara = appendOrReuseParagraph(body, '   ' + sub.subLabel.toLowerCase() + '. ' + cleanTextData(sub.text).trim());
          optPara.setFontSize(11).setFontFamily('Times New Roman').setBold(false).setIndentStart(36);
          optPara.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
        });
        
        if (includeAnswers) {
          var tfStr = q.subQuestions.map(function(s) { return convertToDS(s.answer); }).join('');
          var ansPara = appendOrReuseParagraph(body, '   Đáp án: ' + tfStr);
          ansPara.setFontSize(11).setFontFamily('Times New Roman').setBold(true).setIndentStart(36).setForegroundColor('#000000');
          ansPara.setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
        }
      }
    }

    // HÀM ĐIỀU HƯỚNG PHÂN PHỐI FILE WORD
    function buildAndExportDocx(title, questionList, includeAnswers, examCode, isMixMode) {
      var doc = DocumentApp.create(title);
      var body = doc.getBody();
      body.clear();
      temporaryFileIds.push(doc.getId());
      
      body.setMarginTop(36).setMarginBottom(36).setMarginLeft(36).setMarginRight(36);

      if (isMixMode) {
        var headerTableCells = [
          ['SỞ GDĐT TP. HỒ CHÍ MINH\nTRƯỜNG THPT TÂN PHƯỚC KHÁNH', 'ĐỀ KIỂM TRA ĐÁNH GIÁ NĂNG LỰC\nMôn: Tin học 10\nMã đề: ' + examCode + ' - Thời gian: 45 phút']
        ];
        var headerTable = body.appendTable(headerTableCells);
        headerTable.setBorderWidth(0); 
        
        for (var col = 0; col < 2; col++) {
          var cell = headerTable.getCell(0, col);
          cell.setPaddingTop(0).setPaddingBottom(0).setPaddingLeft(6).setPaddingRight(6);
          cell.getChild(0).asParagraph().setFontFamily('Times New Roman').setFontSize(10).setBold(true).setAlignment(DocumentApp.HorizontalAlignment.CENTER).setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);
        }

        var mcqs = questionList.filter(function(q) { return q.type === 'MCQ'; });
        var tfs = questionList.filter(function(q) { return q.type === 'TF'; });

        if (mcqs.length > 0) {
          var p1Head = appendOrReuseParagraph(body, 'PHẦN I. Câu trắc nghiệm nhiều phương án lựa chọn. Thí sinh trả lời từ câu 1 đến câu ' + mcqs.length + '.');
          p1Head.setFontFamily('Times New Roman').setFontSize(11).setBold(true).setSpacingBefore(6).setSpacingAfter(0).setLineSpacing(1.0);
          
          var p1Sub = appendOrReuseParagraph(body, 'Mỗi câu hỏi thí sinh chỉ chọn một phương án.');
          p1Sub.setFontFamily('Times New Roman').setFontSize(11).setItalic(true).setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);

          mcqs.forEach(function(q, index) {
            renderQuestionBlock(q, index + 1, includeAnswers, true, body);
          });
        }

        if (tfs.length > 0) {
          var p2Head = appendOrReuseParagraph(body, 'PHẦN II. Câu trắc nghiệm đúng sai. Thí sinh trả lời từ câu 1 đến câu ' + tfs.length + '.');
          p2Head.setFontFamily('Times New Roman').setFontSize(11).setBold(true).setSpacingBefore(6).setSpacingAfter(0).setLineSpacing(1.0);
          
          var p2Sub = appendOrReuseParagraph(body, 'Trong mỗi ý a), b), c), d) ở mỗi câu, thí sinh chọn đúng hoặc sai.');
          p2Sub.setFontFamily('Times New Roman').setFontSize(11).setItalic(true).setSpacingBefore(0).setSpacingAfter(0).setLineSpacing(1.0);

          tfs.forEach(function(q, index) {
            renderQuestionBlock(q, index + 1, includeAnswers, true, body);
          });
        }
        
        var pFooter = appendOrReuseParagraph(body, '--- Hết---');
        pFooter.setFontFamily('Times New Roman').setFontSize(11).setBold(false).setAlignment(DocumentApp.HorizontalAlignment.CENTER).setSpacingBefore(12).setSpacingAfter(0).setLineSpacing(1.0);
      } 
      else {
        questionList.forEach(function(q, index) {
          renderQuestionBlock(q, index + 1, includeAnswers, false, body);
        });
      }
      
      doc.saveAndClose();
      
      var exportUrl = "https://docs.google.com/feeds/download/documents/export/Export?id=" + doc.getId() + "&exportFormat=docx";
      var response = UrlFetchApp.fetch(exportUrl, {
        method: "get",
        headers: { "Authorization": "Bearer " + ScriptApp.getOAuthToken() },
        muteHttpExceptions: true
      });
      
      var fileBlob = response.getBlob();
      fileBlob.setContentType("application/vnd.openxmlformats-officedocument.wordprocessingml.document");
      return fileBlob;
    }

    // ================= ĐIỀU PHỐI XUẤT FILE ĐẦU RA =================
    if (mode === 'original') {
      var fileBlob = buildAndExportDocx("DANH SÁCH CÂU HỎI ĐỀ GỐC", questions, false, "GỐC", false);
      fileBlob.setName("De_Thi_Goc.docx");
      var base64Data = Utilities.base64Encode(fileBlob.getBytes());
      cleanUpTemporaryFiles(temporaryFileIds);
      return { success: true, isZip: false, base64: base64Data, fileName: "De_Thi_Goc.docx" };
    }

    if (mode === 'with_answers') {
      var fileBlob = buildAndExportDocx("ĐỀ THI GỐC KÈM ĐÁP ÁN CHI TIẾT", questions, true, "ĐÁP ÁN", false);
      fileBlob.setName("De_Thi_Goc_Kem_Dap_An.docx");
      var base64Data = Utilities.base64Encode(fileBlob.getBytes());
      cleanUpTemporaryFiles(temporaryFileIds);
      return { success: true, isZip: false, base64: base64Data, fileName: "De_Thi_Goc_Kem_Dap_An.docx" };
    }

    if (mode === 'mix') {
      var zipBlobs = [];
      var excelMatrix = []; 
      
      for (var c = 0; c < codeCount; c++) {
        var currentCode = startCode + c;
        
        var mcqs = questions.filter(function(q) { return q.type === 'MCQ'; }).map(function(q) { return JSON.parse(JSON.stringify(q)); });
        var tfs = questions.filter(function(q) { return q.type === 'TF'; }).map(function(q) { return JSON.parse(JSON.stringify(q)); });
        
        shuffleArray(mcqs);
        mcqs.forEach(function(q) {
          if (q.options && q.options.length > 0) {
            var originalOpts = q.options.map(function(opt, idx) { return { text: opt, isCorrect: idx === q.correctAnswer }; });
            shuffleArray(originalOpts);
            q.options = originalOpts.map(function(o) { return o.text; });
            var newCorrect = originalOpts.findIndex(function(o) { return o.isCorrect; });
            q.correctAnswer = newCorrect !== -1 ? newCorrect : null;
          }
        });
        
        shuffleArray(tfs);
        var mixedExam = mcqs.concat(tfs);
        
        var fileBlob = buildAndExportDocx("ĐỀ KIỂM TRA ĐÁNH GIÁ NĂNG LỰC", mixedExam, false, currentCode, true);
        fileBlob.setName("De_Thi_Ma_De_" + currentCode + ".docx");
        zipBlobs.push(fileBlob);
        
        var codeAnswers = [];
        mixedExam.forEach(function(q) {
          if (q.type === 'MCQ') {
            var letter = q.correctAnswer !== null ? String.fromCharCode(65 + q.correctAnswer) : "A";
            codeAnswers.push(letter);
          } else if (q.type === 'TF') {
            var tfStr = q.subQuestions.map(function(s) { return convertToDS(s.answer); }).join('');
            codeAnswers.push(tfStr);
          }
        });
        excelMatrix.push({ code: currentCode, answers: codeAnswers });
      }
      
      var csvContent = "\uFEFF" + "Câu hỏi,Dạng câu hỏi";
      for (var i = 0; i < codeCount; i++) { csvContent += ",Mã đề " + (startCode + i); }
      csvContent += "\n";
      
      var totalQ = excelMatrix[0].answers.length;
      for (var qIdx = 0; qIdx < totalQ; qIdx++) {
        var rowText = "Câu " + (qIdx + 1) + ",";
        var sampleAns = excelMatrix[0].answers[qIdx];
        rowText += (sampleAns.length === 4) ? "Đúng/Sai (DSSD)" : "Trắc nghiệm";
        
        for (var cIdx = 0; cIdx < codeCount; cIdx++) {
          rowText += "," + excelMatrix[cIdx].answers[qIdx];
        }
        csvContent += rowText + "\n";
      }
      
      var csvBlob = Utilities.newBlob(csvContent, "text/csv; charset=utf-8", "Bang_Dap_An_Tong_Hop.csv");
      zipBlobs.push(csvBlob);

      var zipBlob = Utilities.zip(zipBlobs, "Go_De_Thi_SmartExam_Tron_Goi.zip");
      var base64Data = Utilities.base64Encode(zipBlob.getBytes());
      
      cleanUpTemporaryFiles(temporaryFileIds);
      return { success: true, isZip: true, base64: base64Data, fileName: "Go_De_Thi_SmartExam_Tron_Goi.zip" };
    }

  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

function shuffleArray(array) {
  if (!array || array.length <= 1) return;
  for (var i = array.length - 1; i > 0; i--) {
    var j = Math.floor(Math.random() * (i + 1));
    var temp = array[i];
    array[i] = array[j];
    array[j] = temp;
  }
}

function cleanUpTemporaryFiles(fileIds) {
  fileIds.forEach(function(id) {
    try { DriveApp.getFileById(id).setTrashed(true); } catch(e) {}
  });
}
